//CREAR PROYECTO
1: instalar ionic (-g/global)
npm install -g @ionic/cli

2: crear proyecto
ionic start proyecto1 ?template_a_usar --type=react
ionic start plantilla1 sidemenu --type=react
ionic start plantilla1_cordova sidemenu --type=react
ionic start app_web sidemenu --type=react

3: integrar capasitor
cd nombre_proyecto
ionic integrations enable capacitor ... activar
ionic integrations disable capacitor ... desactivar

4: integrar android e ios
ionic capacitor add android
ionic capacitor add ios
